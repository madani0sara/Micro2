const int ledpin = 8;

void setup() {
  pinMode(ledpin, OUTPUT);
  Serial.begin (9600);
}

  void loop() {
    if (Serial.available() >0){
    char incomingByte = Serial.read();
      if (incomingByte == 'H') {
        digitalWrite(ledpin, HIGH);

      }  else if (incomingByte == 'L')  {
        digitalWrite (ledpin, LOW);
      }
    }
  }
